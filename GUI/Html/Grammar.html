Грамматика

Определим грамматику объявления списка с инициализацией на языке Kotlin G[Z] в нотации Хомского с продукциями P: 

<declaration> -> "val" <identifier> "=" "listOf" "(" <elements_opt> ")" ";"

<elements_opt> -> <elements>
<elements_opt> -> ε

<elements> -> <element>
<elements> -> <element> "," <elements>

<element> -> <string>
<element> -> <char>
<element> -> "true"
<element> -> "false"
<element> -> <number>

<number> -> <int>
<number> -> <double>
<number> -> <signed_number>

<signed_number> -> <sign> <int>
<signed_number> -> <sign> <double>

<sign> -> "+"
<sign> -> "-"

<identifier> -> <letter> <identifier_tail>
<identifier> -> "_" <identifier_tail>

<identifier_tail> -> <letter> <identifier_tail>
<identifier_tail> -> <digit> <identifier_tail>
<identifier_tail> -> "_" <identifier_tail>
<identifier_tail> -> ε

<string> -> "\"" <string_body> "\""

<string_body> -> <string_char> <string_body>
<string_body> -> ε

<string_char> -> <letter>
<string_char> -> <digit>
<string_char> -> "_"
<string_char> -> " "

<char> -> "'" <char_symbol> "'"

<char_symbol> -> <letter>
<char_symbol> -> <digit>
<char_symbol> -> "_"
<char_symbol> -> " "

<int> -> <digit> <digits>

<double> -> <digit> <digits> "." <digit> <digits>

<digits> -> <digit> <digits>
<digits> -> ε

<digit> -> "0"
<digit> -> "1"
<digit> -> "2"
<digit> -> "3"
<digit> -> "4"
<digit> -> "5"
<digit> -> "6"
<digit> -> "7"
<digit> -> "8"
<digit> -> "9"

<letter> -> "A" | "B" | ... | "Z" | "a" | "b" | ... | "z"
